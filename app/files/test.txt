Enum job_type {
  freelancer
  employee
}

Table work_histories {
  id bigint [pk]
  user_id bigint [not null, ref: > users.id]
  seq int [not null]
  project_name string [not null]
  start_date date [not null]
  role int [not null]
  team_size char(3) [not null]
  responsible_process json
  project_detail text
}

Table skills {
  id bigint [pk]
  user_id bigint [ref: > users.id]
  seq bigint [not null]
  m_skill_id bigint [ref: > m_skills.id]
}
